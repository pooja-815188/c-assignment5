#include<stdio.h>
int missingnum(int arr[],int n)
{
    int m=n+1;
    int total=m*(m+1)/2;
    int sum=0;
    for(int i=0;i<n;i++)
    {
        sum=sum+arr[i];
    }
    return total-sum;
}
int main()
{
    int arr[]={1,2,3,5,6,7,8,9};
    int n=sizeof(arr)/sizeof(arr[0]);
    
    printf("%d\n",missingnum(arr,n));
}
