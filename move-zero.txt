#include <stdio.h>
void move_zeros_to_end(int *arr,int size)
{
   int i,temp,count=1;
    for(i=0; i<size; i++)
    {
        if(arr[i] == 0 && arr[size-count] != 0)
        {
            temp = arr[i];
            arr[i] = arr[size-count];
            arr[size-count] = temp;
            count++;

        }
    }
    printf("Array contains:\n");
    for(i=0; i<size; i++)
    {
        printf("%d ",*(arr+i));
    }
    printf("\n");
}

int main()
{
    int arr[]= {1,3,0,4,6,0,9};
    int size = sizeof(arr)/sizeof(arr[0]);
    move_zeros_to_end(arr,size);
    return 0;
}
